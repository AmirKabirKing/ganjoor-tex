\begin{center}
\section*{شماره ۲۶: هر لحظه ز چرخ بیش میباید رفت}
\label{sec:026}
\addcontentsline{toc}{section}{\nameref{sec:026}}
\begin{longtable}{l p{0.5cm} r}
هر لحظه ز چرخ بیش میباید رفت
&&
گاه از بس و گه زپیش میباید رفت
\\
در گردِ جهان دویدنت فایده نیست
&&
گردِ سر و پای خویش میباید رفت
\\
\end{longtable}
\end{center}
