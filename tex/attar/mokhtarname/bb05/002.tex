\begin{center}
\section*{شماره ۲: چون ما به وجود خود هویدا باشیم}
\label{sec:002}
\addcontentsline{toc}{section}{\nameref{sec:002}}
\begin{longtable}{l p{0.5cm} r}
چون ما به وجود خود هویدا باشیم
&&
چون ما به وجود خود هویدا باشیم
\\
تو هیچ نهیی ولیک میپنداری
&&
تو هیچ نهیی ولیک میپنداری
\\
\end{longtable}
\end{center}
