\begin{center}
\section*{دوبیتی شماره ۶۱: فلک زار و نزارم کردی آخر}
\label{sec:061}
\addcontentsline{toc}{section}{\nameref{sec:061}}
\begin{longtable}{l p{0.5cm} r}
فلک زار و نزارم کردی آخر
&&
جدا از گلعذارم کردی آخر
\\
میان تختهٔ نرد محبت
&&
شش و پنجی بکارم کردی آخر
\\
\end{longtable}
\end{center}
