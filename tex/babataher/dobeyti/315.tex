\begin{center}
\section*{دوبیتی شماره ۳۱۵: سیه بختم که بختم واژگون بی}
\label{sec:315}
\addcontentsline{toc}{section}{\nameref{sec:315}}
\begin{longtable}{l p{0.5cm} r}
سیه بختم که بختم واژگون بی
&&
سیه روجم که روجم سرنگون بی
\\
شدم آوارهٔ کوی محبت
&&
زدست دل که یارب غرق خون بی
\\
\end{longtable}
\end{center}
